Complete Facebook SDK Code (Ready to Use)
Copy and paste this entire code block into your login page HTML. It should go right after the opening <body> tag or at the very top of your <head> section:
html<script>
  window.fbAsyncInit = function() {
    FB.init({
      appId      : '136830059484019',
      cookie     : true,
      xfbml      : true,
      version    : 'v18.0'
    });
    FB.AppEvents.logPageView();
  };
  
  (function(d, s, id){
    var js, fjs = d.getElementsByTagName(s)[0];
    if (d.getElementById(id)) {return;}
    js = d.createElement(s);
    js.id = id;
    js.src = "https://connect.facebook.net/en_US/sdk.js";
    fjs.parentNode.insertBefore(js, fjs);
  }(document, 'script', 'facebook-jssdk'));
</script>
