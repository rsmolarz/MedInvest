document.addEventListener('DOMContentLoaded', function() {
    const expenses = document.getElementById('expenses');
    const swr = document.getElementById('swr');
    const result = document.getElementById('freedomNumber');
    function calculate() {
        const monthlyExp = parseFloat(expenses.value) || 0;
        const rate = parseFloat(swr.value) || 0.04;
        const freedomNum = (monthlyExp * 12) / rate;
        result.textContent = '$' + freedomNum.toLocaleString('en-US', {maximumFractionDigits: 0});
    }
    expenses.addEventListener('input', calculate);
    swr.addEventListener('change', calculate);
});
